      hotel_avg_scores = df.groupby('Hotel_Name')['Reviewer_Score'].mean().sort_values()
      worst_hotels = hotel_avg_scores.head(10)
      best_hotels = hotel_avg_scores.tail(10)
      plt.figure(figsize=(16, 6))
      # Worst Hotels
      plt.subplot(1, 2, 1)
      worst_hotels.plot(kind='barh', color='crimson')
      plt.title('Worst 10 Hotels by Reviewer Score')
      plt.xlabel('Average Reviewer Score')
      plt.gca().invert_yaxis()
      # Best Hotels
      plt.subplot(1, 2, 2)
      best_hotels.plot(kind='barh', color='seagreen')
      plt.title('Best 10 Hotels by Reviewer Score')
      plt.xlabel('Average Reviewer Score')
      plt.gca().invert_yaxis()
      plt.tight_layout()
